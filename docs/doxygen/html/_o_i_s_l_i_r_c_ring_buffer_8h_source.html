76 class LIRCRingBuffer
77 {
78 private:
80 int bufferSize;
82 int bigMask;
83 // Used for fitting indices to buffer.
84 int smallMask;
85
86 // Buffer holding the actual event buffers
87 LIRCEvent* buffer;
88
90 volatile int writeIndex;
91
93 volatile int readIndex;
94
95 public:
96 LIRCRingBuffer(unsigned int numEntries)
97 {
98 numEntries = RoundUpToNextPowerOf2(numEntries);
99
100 //2 bytes per short
101 bufferSize = (int)numEntries;
102 buffer = new LIRCEvent[numEntries];
103
104 Flush();
105
106 bigMask = (int)(numEntries * 2) - 1;
107 smallMask = (int)(numEntries)-1;
108 }
109
110 ~LIRCRingBuffer()
111 {
112 delete buffer;
113 }

115 unsigned int RoundUpToNextPowerOf2(unsigned int n)
116 {
117 int numBits = 0;
118 if(((n - 1) & n) == 0)
119 return n; //Already Power of two.
120
121 while(n > 0)
122 {
123 n = n >> 1;
124 numBits++;
125 }
126 return (unsigned int)(1 << numBits);
127 }
128
129 int GetReadAvailable()
130 {
131 return ((writeIndex - readIndex) & bigMask);
132 }
133
134 int GetWriteAvailable()
135 {
136 return (bufferSize - GetReadAvailable());
137 }
138
139 int Write(LIRCEvent* data, int numEntries)
140 {
141 int size1 = 0, size2 = 0, numWritten;
142 int data1Ptr = 0, data2Ptr = 0;
143
144 numWritten = GetWriteRegions(numEntries, data1Ptr, size1, data2Ptr, size2);
145
146 if(size2 > 0)
147 {
148 //copy to two parts
149 memcpy(&buffer[data1Ptr], data, sizeof(LIRCEvent) * size1);
150 //Array.Copy( data, offsetPtr, buffer, data1Ptr, size1 );
151 memcpy(&buffer[data2Ptr], &data[size1], sizeof(LIRCEvent) * size2);
152 //Array.Copy( data, offsetPtr + size1, buffer, data2Ptr, size2 );
153 }
154 else
155 { //Copy all continous
156 memcpy(&buffer[data1Ptr], data, sizeof(LIRCEvent) * size1);
157 //Array.Copy( data, offsetPtr, buffer, data1Ptr, size1 );
158 }
159 AdvanceWriteIndex(numWritten);
160 return numWritten;
161 }
162
167 int Read(LIRCEvent* data, int numEntries)
168 {
169 int size1 = 0, size2 = 0, numRead, data1Ptr = 0, data2Ptr = 0;
170
171 numRead = GetReadRegions(numEntries, data1Ptr, size1, data2Ptr, size2);
172
173 if(size2 > 0)
174 {
175 memcpy(data, &buffer[data1Ptr], sizeof(LIRCEvent) * size1);
176 //Array.Copy( buffer, data1Ptr, data, 0, size1 );
177 memcpy(&data[size1], &buffer[data2Ptr], sizeof(LIRCEvent) * size2);
178 //Array.Copy( buffer, data2Ptr, data, size1, size2 );
179 }
180 else
181 memcpy(data, &buffer[data1Ptr], sizeof(LIRCEvent) * size1);
182 //Array.Copy( buffer, data1Ptr, data, 0, size1 );
183
184 AdvanceReadIndex(numRead);
185 return numRead;
186 }
187
188 private:
189 int GetWriteRegions(int numEntries, int& dataPtr1, int& sizePtr1, int& dataPtr2, int& sizePtr2)
190 {
191 int index;
192 int available = GetWriteAvailable();
193 if(numEntries > available)
194 numEntries = available;
195
196 //Check to see if write is not contiguous.
197 index = writeIndex & smallMask;
198 if((index + numEntries) > bufferSize)
199 {
200 //Write data in two blocks that wrap the buffer.
201 int firstHalf = bufferSize - index;
202 dataPtr1 = index; //&buffer[index];
203 sizePtr1 = firstHalf;
204 dataPtr2 = 0; //&buffer[0];
205 sizePtr2 = numEntries - firstHalf;
206 }
207 else
208 {
209 dataPtr1 = index; //&buffer[index];
210 sizePtr1 = numEntries;
211 dataPtr2 = 0;
212 sizePtr2 = 0;
213 }
214 return numEntries;
215 }
216
217 int GetReadRegions(int numEntries, int& dataPtr1, int& sizePtr1, int& dataPtr2, int& sizePtr2)
218 {
219 int index;
220 int available = GetReadAvailable();
221 if(numEntries > available)
222 numEntries = available;
223
224 // Check to see if read is not contiguous
225 index = readIndex & smallMask;
226 if((index + numEntries) > bufferSize)
227 {
228 // Write data in two blocks that wrap the buffer
229 int firstHalf = bufferSize - index;
230 dataPtr1 = index; //&buffer[index];
231 sizePtr1 = firstHalf;
232 dataPtr2 = 0; //&buffer[0];
233 sizePtr2 = numEntries - firstHalf;
234 }
235 else
236 {
237 dataPtr1 = index; //&buffer[index];
238 sizePtr1 = numEntries;
239 dataPtr2 = 0;
240 sizePtr2 = 0;
241 }
242 return numEntries;
243 }
244
245 int AdvanceWriteIndex(int numEntries)
246 {
247 return writeIndex = (writeIndex + numEntries) & bigMask;
248 }
249
250 int AdvanceReadIndex(int numEntries)
251 {
252 return readIndex = (readIndex + numEntries) & bigMask;
253 }

255 void Flush()
256 {
257 writeIndex = readIndex = 0;
258 }
259 };
