112  unsigned int RoundUpToNextPowerOf2( unsigned int n )
113  {
114  int numBits = 0;
115  if( ((n-1) & n) == 0)
116  return n; //Already Power of two.
117 
118  while( n > 0 )
119  {
120  n= n>>1;
121  numBits++;
122  }
123  return (unsigned int)(1<<numBits);
