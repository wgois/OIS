182 class _OISExport Envelope : public ForceEffect
183 {
184 public:
185 Envelope() :
186 attackLength(0), attackLevel(0), fadeLength(0), fadeLevel(0) { }
187#if defined(OIS_MSVC_COMPILER)
188#pragma warning(push)
189#pragma warning(disable : 4800)
190#endif
191 bool isUsed() const
192 {
193 return attackLength | attackLevel | fadeLength | fadeLevel;
194 }
195#if defined(OIS_MSVC_COMPILER)
196#pragma warning(pop)
197#endif
198
199 // Duration of the attack (microseconds)
200 unsigned int attackLength;
201
202 // Absolute level at the beginning of the attack (0 to 10K)
203 // (automatically signed when necessary by FF core according to effect level sign)
204 unsigned short attackLevel;
205
206 // Duration of fade (microseconds)
207 unsigned int fadeLength;
208
209 // Absolute level at the end of fade (0 to 10K)
210 // (automatically signed when necessary by FF core according to effect level sign)
211 unsigned short fadeLevel;
212 };
