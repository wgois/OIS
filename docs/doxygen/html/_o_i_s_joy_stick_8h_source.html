71 class _OISExport JoyStickState
72 {
73 public:
75 JoyStickState() { clear(); }
76
78 std::vector<bool> mButtons;
79
81 std::vector<Axis> mAxes;
82
84 Pov mPOV[4];
85
87 Slider mSliders[4];
88
90 std::vector<Vector3> mVectors;
91
93 void clear()
94 {
95 for(std::vector<bool>::iterator i = mButtons.begin(), e = mButtons.end(); i != e; ++i)
96 {
97 (*i) = false;
98 }
99
100 for(std::vector<Axis>::iterator i = mAxes.begin(), e = mAxes.end(); i != e; ++i)
101 {
102 i->absOnly = true; //Currently, joysticks only report Absolute values
103 i->clear();
104 }
105
106 for(std::vector<Vector3>::iterator i = mVectors.begin(), e = mVectors.end(); i != e; ++i)
107 {
108 i->clear();
109 }
110
111 for(int i = 0; i < 4; ++i)
112 {
113 mPOV[i].direction = Pov::Centered;
114 mSliders[i].abX = mSliders[i].abY = 0;
115 }
116 }
117 };
